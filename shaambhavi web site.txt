<script>
  const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbwpZF63yU6JSaE5HksoN-8-yjBPM1ibwUqvEOKay2lJp8BR9tfvY9haqQcuTTzfRIou/exec";

  // Live open/closed status (IST: Mon-Sat 10:00-19:00)
  (function updateStatus() {
    const now = new Date(new Date().toLocaleString('en-US', {timeZone:'Asia/Kolkata'}));
    const day = now.getDay();   // 0=Sun, 6=Sat
    const h = now.getHours(), m = now.getMinutes();
    const mins = h * 60 + m;
    const isWorkday = day >= 1 && day <= 6;
    const isHours = mins >= 600 && mins < 1140; // 10:00–19:00
    const isOpen = isWorkday && isHours;
    const badge = document.getElementById('liveStatus');
    const txt = document.getElementById('statusText');
    if (!isOpen) {
      badge.style.background = 'rgba(107,114,128,0.12)';
      badge.style.borderColor = 'rgba(107,114,128,0.25)';
      badge.querySelector('.status-dot-inner').style.background = '#6b7280';
      badge.querySelector('.status-dot-inner').style.animation = 'none';
      badge.querySelector('.status-dot-outer').style.background = 'rgba(107,114,128,0.25)';
      txt.innerHTML = day === 0
        ? '<strong style="color:#9ca3af">Closed Today (Sunday)</strong> — Opens Monday at 10:00 AM'
        : '<strong style="color:#9ca3af">Closed Now</strong> — Opens Mon–Sat: 10:00 AM – 7:00 PM';
    }
  })();

  // Product data with scale factor (0-1) controlling rendered bottle height
  const products = [

  /**
   * Draws a realistic aluminium bottle SVG that scales proportionally.
   * scale: 0..1 — controls rendered height (smaller = shorter bottle)
   * wide: true for large jerry-can style bottles
   * modalSize: true for larger modal rendering
   */
  function bottleSVG(p, modalSize) {
    const scale = p.scale;
    const isWide = !!p.wide;
    const uid = 'b' + Math.random().toString(36).slice(2,8);

    /* ---- Canvas dimensions ---- */
    const VW = modalSize ? 100 : 80;   // viewBox width
    const VH = modalSize ? 200 : 130;  // viewBox height

    /* ---- Bottle proportions ---- */
    // Body width: small bottles slim, large bottles wide
    const minBodyW = isWide ? 42 : 18;
    const maxBodyW = isWide ? 62 : 46;
    const bodyW = minBodyW + (maxBodyW - minBodyW) * scale;

    // Body height fills most of VH, scaled
    const minBodyH = 20;
    const maxBodyH = VH * (isWide ? 0.60 : 0.68);
    const bodyH = minBodyH + (maxBodyH - minBodyH) * scale;

    // Shoulder: tapered top of body
    const shoulderH = bodyW * (isWide ? 0.18 : 0.28);

    // Neck width & height
    const neckW = isWide ? bodyW * 0.25 : Math.max(8, bodyW * 0.35);
    const neckH = isWide ? 6 : Math.max(6, bodyH * 0.12);

    // Cap
    const capW = neckW + 3;
    const capH = isWide ? 8 : Math.max(5, neckH * 0.55);

    // Total height of bottle
    const totalH = capH + neckH + shoulderH + bodyH;
    // Top Y so bottle sits at BOTTOM of viewBox
    const topY = VH - totalH;

    const cx = VW / 2;

    // X positions
    const capX = cx - capW / 2;
    const neckX = cx - neckW / 2;
    const bodyX = cx - bodyW / 2;

    /* ---- Gradients ---- */
    // Aluminium brushed metal gradient (left-to-right)
    const metalGrad = `
      <linearGradient id="${uid}m" x1="0%" y1="0%" x2="100%" y2="0%">
        <stop offset="0%"   stop-color="#5a6270"/>
        <stop offset="10%"  stop-color="#9ba3af"/>
        <stop offset="28%"  stop-color="#d4d8de"/>
        <stop offset="42%"  stop-color="#f0f2f4"/>
        <stop offset="50%"  stop-color="#ffffff"/>
        <stop offset="60%"  stop-color="#e8eaec"/>
        <stop offset="75%"  stop-color="#b0b7c0"/>
        <stop offset="90%"  stop-color="#7c8592"/>
        <stop offset="100%" stop-color="#4a5260"/>
      </linearGradient>`;

    // Cap: white plastic look
    const capGrad = `
      <linearGradient id="${uid}c" x1="0%" y1="0%" x2="100%" y2="0%">
        <stop offset="0%"   stop-color="#c8cacb"/>
        <stop offset="30%"  stop-color="#f0f1f2"/>
        <stop offset="50%"  stop-color="#ffffff"/>
        <stop offset="75%"  stop-color="#dddfe0"/>
        <stop offset="100%" stop-color="#b0b2b3"/>
      </linearGradient>`;

    // Shoulder: slightly darker gradient
    const shoulderGrad = `
      <linearGradient id="${uid}s" x1="0%" y1="0%" x2="100%" y2="0%">
        <stop offset="0%"   stop-color="#4a5260"/>
        <stop offset="20%"  stop-color="#8e96a2"/>
        <stop offset="45%"  stop-color="#d8dce2"/>
        <stop offset="55%"  stop-color="#f4f5f6"/>
        <stop offset="78%"  stop-color="#9ba3af"/>
        <stop offset="100%" stop-color="#3e4656"/>
      </linearGradient>`;

    /* ---- Shoulder path: trapezoid from neckX..neckX+neckW at top to bodyX..bodyX+bodyW at bottom ---- */
    const sY = topY + capH + neckH;
    const shoulderPath = `M ${neckX} ${sY} L ${neckX + neckW} ${sY} L ${bodyX + bodyW} ${sY + shoulderH} L ${bodyX} ${sY + shoulderH} Z`;

    /* ---- Wide bottle handle (25L style) ---- */
    const handleSVG = isWide && p.scale > 0.95
      ? `<path d="M ${cx-10} ${topY+2} Q ${cx-10} ${topY-10} ${cx} ${topY-10} Q ${cx+10} ${topY-10} ${cx+10} ${topY+2}"
           fill="none" stroke="url(#${uid}m)" stroke-width="5" stroke-linecap="round"/>`
      : '';

    /* ---- Cap grip rings (horizontal lines on cap) ---- */
    const ringCount = Math.max(2, Math.round(capH / 3));
    let rings = '';
    for (let i = 1; i < ringCount; i++) {
      const ry = topY + (i / ringCount) * capH;
      rings += `<line x1="${capX+1}" y1="${ry}" x2="${capX+capW-1}" y2="${ry}" stroke="rgba(0,0,0,0.1)" stroke-width="0.6"/>`;
    }

    /* ---- Highlight stripe on body ---- */
    const hlX = bodyX + bodyW * 0.16;
    const hlW = Math.max(3, bodyW * 0.10);
    const hlY1 = sY + shoulderH + bodyH * 0.06;
    const hlY2 = sY + shoulderH + bodyH * 0.88;

    /* ---- Bottom rounded edge ---- */
    const bottomY = sY + shoulderH + bodyH;
    const bottomRY = Math.min(5, bodyW * 0.06);

    /* ---- Label rectangle ---- */
    const labelY = sY + shoulderH + bodyH * 0.22;
    const labelH = bodyH * 0.38;
    const labelX = bodyX + bodyW * 0.06;
    const labelW = bodyW * 0.88;

    return `<svg viewBox="0 0 ${VW} ${VH}" xmlns="http://www.w3.org/2000/svg" width="${VW}" height="${VH}">
      <defs>
        ${metalGrad}
        ${capGrad}
        ${shoulderGrad}
      </defs>

      ${handleSVG}

      <!-- Cap -->
      <rect x="${capX}" y="${topY}" width="${capW}" height="${capH}" rx="${Math.min(3, capH/2)}"
            fill="url(#${uid}c)"/>
      ${rings}
      <!-- Cap top shine -->
      <rect x="${capX+2}" y="${topY+1}" width="${capW-4}" height="${Math.max(1.5, capH*0.25)}" rx="1"
            fill="rgba(255,255,255,0.55)"/>

      <!-- Neck -->
      <rect x="${neckX}" y="${topY+capH}" width="${neckW}" height="${neckH}" rx="1.5"
            fill="url(#${uid}m)"/>

      <!-- Shoulder -->
      <path d="${shoulderPath}" fill="url(#${uid}s)"/>

      <!-- Body -->
      <rect x="${bodyX}" y="${sY+shoulderH}" width="${bodyW}" height="${bodyH}"
            rx="${Math.min(6, bodyW*0.08)}" fill="url(#${uid}m)"/>

      <!-- Bottom ellipse shadow -->
      <ellipse cx="${cx}" cy="${bottomY}" rx="${bodyW/2}" ry="${bottomRY}"
               fill="rgba(0,0,0,0.18)"/>

      <!-- Body highlight stripe -->
      <rect x="${hlX}" y="${hlY1}" width="${hlW}" height="${hlY2-hlY1}" rx="${hlW/2}"

    </svg>`;
  }
